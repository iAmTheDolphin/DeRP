unary
	: INT
	| REAL
	| STRING
	| OPAREN expression CPAREN
	| var_expr
	| NOT unary

var_expr
    : ID
    | ID OPAREN opt_param_list CPAREN
    | ID OBRACKET UNARY CBRACKET

operator
	: PLUS
	| MINUS
	| TIMES
	| DIVIDES
	| ASSIGN

expression
    : unary_expr
    | if
    | loop
    | otherwise

unary_expr
    : unary
    | unary operator unary
    | unary comp unary

expr_list
	: expression
	| expression expr_list

opt_expr_list
	: expression opt_expr_list
	| EMPTY

defs
    : function_def
	| array_def
	| var_def

 param_list
	: unary
	| unary COMMA paramList

opt_param_list
	: param_list
	| EMPTY

function_def
	: FUNCTION ID USING OPAREN opt_param_list CPAREN

body
	: OBRACE opt_expr_list CBRACE

array_def
	: LIST unary OBRACKET UNARY CBRACKET

var_def
	: VAR ID EQUALS unary

var_assign
	: ID EQUALS unary

comp
	: GREATER
	| LESS
	| GREATEREQUAL
	| GREATERLESS
	| NOTEQUAL

loop :
	| while

while
	: LOOP WHILE condition_list body
	| LOOP WHILE condition_list expression

if
	: IF condition_list expression opt_otherwise
	| IF condition_list body opt_otherwise

opt_otherwise
    : otherwise
    | EMPTY

otherwise
	: OTHERWISE expression
	| OTHERWISE if

linker
	: AND
	| OR
	| NOT

condition_list
	: expression
	| expression linker condition_list

condition
    : expression comp expression

return
	: RETURN expression

array_assign
	: UNARY OBRACKET UNARY CBRACKET EQUALS expression

rement
	: INCREMENT unary
	| DECREMENT unary

program
    	: opt_func_def_list

opt_func_def_list
    	: func_def_list
    	| EMPTY

func_def_list
    	: function_def func_def_list
    	| function_def
