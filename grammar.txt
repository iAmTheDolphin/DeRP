
1: unary 
	: INT
	| REAL
	| STRING
	| OPAREN expression CPAREN
	| var_expr
	| NOT unary

var_expr
    	: ID
    	| ID OPAREN opt_param_list CPAREN
    	| UNARY OBRACKET UNARY CBRACKET

2: operator
	: PLUS
	| MINUS
	| TIMES
	| DIVIDE

expression
    	: unary_expr
    	| if
    	| loop
    	| otherwise


unary_expr
    	: unary
    	| unary operator unary
    	| unary comp unary

expr_list
	: expression
	| opt_expr_list

opt_expr_list
	: expression opt_expr_list
	| EMPTY


defs
: function_def
	| array_def
	| var_def

 param_list
	: unary
	| unary COMMA paramList

7: opt_param_list
	: param_list
	| EMPTY

10: function_def
	: FUNCTION ID USING OPAREN opt_param_list CPAREN

11: body
	: OBRACE opt_expr_list CBRACE

12: array_def
	: LIST UNARY OBRACKET UNARY CBRACKET

14: var_def
	: VAR ID EQUALS UNARY

15: var_assign
	: ID EQUALS unary


17: comp
	: GREATER
	| LESS
	| GREATEREQUAL
	| GREATERLESS

loop :
	| while

18: while
	: LOOP WHILE condition_list body

19: if
	: IF condition_list expression
	| IF condition_list body

20: otherwise
	: OTHERWISE expression
	| OTHERWISE if

21: linker
	: AND
	| OR

22: condition_list
	: expression comp expression
	| expression comp expression linker condition_list

23: return
	: RETURN expression

24: array_assign
	: UNARY OBRACKET UNARY CBRACKET EQUALS expression

25: rement
	: INCREMENT unary
	| DECREMENT unary

program
    	: opt_func_def_list

opt_func_def_list
    	: func_def_list
    	| EMPTY

func_def_list
    	: function_def func_def_list
    	| function_def
